        _C(exchange.Buy, price, amount)

    def swapCloseBuy(self, price, count):
        _C(exchange.SetDirection, "closebuy")
        _C(exchange.Sell, price, count)

    def swapSell(self, price, count):
        _C(exchange.SetDirection, "sell")
        _C(exchange.Sell, price, count)

    def swapCloseSell(self, price, count):
        _C(exchange.SetDirection, "closesell")
        _C(exchange.Buy, price, count)
